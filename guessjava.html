            if(userInput == compNumber){

            }
            else if (userInput > compNumber){
                out.println("Try again! The Number is Lesser then the Number You've guessed.");
            }
            else if (userInput < compNumber){
                out.println("Try again! The Number is Greater than the Number you've guessed.");
            }
            count = count + 1;
        };
        out.println("you've guessed it right in " +count + " attempts");
