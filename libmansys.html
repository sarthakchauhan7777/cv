    class Library:
    def __init__(self,listofbooks):
        self.books = listofbooks
    def displaybooks(self):
        print("The books present in this library are:- ")
        for book in self.books:
            print("\t*" + book)

    def borrowbook(self,bookName,name,days_of_borrow):
        if bookName in self.books:
            self.books.remove(bookName)
            self.name = name
            self.days_of_borrow = days_of_borrow
            print(f"The book {bookName} has been issued to you for {days_of_borrow} days.")
            with open ("library borrower file.txt",'a') as f:
                f.write(f"\n{bookName} is issued to {name} for {days_of_borrow} days.")
            return True
        else:
            print("SORRY: The book you mentioned either not present in the library\nor the book is unavailable at the moment please! Try again after some days")
            return False
