Source Code for the Stone Paper Scissors Game 

    import random

def game(comp, a):
    if comp ==a :
        return None
    if comp == "a" :
        if a == "p":
            return True
        elif a=="c":
            return False
    if comp == "b" :
        if a == "c":
            return True
        elif a=="s":
            return False
    if comp == "c" :
        if a == "s":
            return True
        elif a == "p":
            return False

def compchoice(comp)   :
    if comp == "a":
        return("computer chose for stone")
    elif comp== "b":
        return("computer chose for paper")
    else :
        return("computer chose for scissors")

def yourchoice(a) :
    if a == "s":
        return("you chose for stone")
    elif a=="p":
        return ("you chose for paper")
    elif a == "c":
        return ("you chose for scissors")


print("computer's turn choose your move stone(1),paper(2) and scissors(3)")

    comp = "a"
